\[
   (\fun x.\fun y.(1+x)+4) ((\fun z.z*z)2)
\]


\begin{align*}
  \textsl{CBV fuerte y debil}\\
  & (\fun x.\fun y.(1+x)+4) ((\fun z.z*z)2)\\
  & \to (\fun x.\fun y.(1+x)+4) (2*2)\\
  & \to (\fun x.\fun y.(1+x)+4) 4\\
  & \to \fun y.(1+4)+4\\
  & \to \fun y.~5+4\\
  & \to \fun y.~9\\
  & \to\fbox{9}\\	
\end{align*}
\begin{align*}
	\textsl{CBN fuerte y debil}\\
  & (\fun x.\fun y.(1+x)+4) ((\fun z.z*z)2)\\
  & \to \fun y.(1+((\fun z.z*z)2))+4\\
  & \to (1+((\fun z.z*z)2))+4\\
  & \to (1+(2*2))+4\\
  & \to (1+4)+4\\
  & \to 5+4\\
  & \to\fbox{9}\\
\end{align*}
