	\textsl{Extender PCF para disponer de Arboles Binarios} \\

	\textbf{Sintaxis de Tipos: } \\
	
	  $ A = \Nat | A\Rightarrow A | A\wedge A | Tree~A $ \\
	
	\textbf{Sintaxis de terminos:  } \\
	 
	$ t = ... | Leaf~t | Tree~t~t~t | ifLeaf~t\Then t\Else t | CL~t | CR~t | Node~t $ \\
	
	\textbf{Semantica Operacional:} \\
	
	$ \CL \Tree t~u~v\rightarrow u $  \\
	$ \CR \Tree t~u~v\rightarrow v $  \\
	$ \ifLeaf \Tree t~u~v\Then u\Else v\rightarrow v $  \\
	$ \ifLeaf \Leaf t\Then u\Else v\rightarrow u $  \\
	$ \Node \Tree t~u~v\rightarrow t $  \\
	$ \Node \Leaf t\rightarrow t $  \\
	
	
	\textbf{Relacion definida inductivamente:} \\
	
	\infer{\Gamma \vdash \Leaf t: A \Rightarrow \Leaf A}{\Gamma \vdash t : A}
	\vspace{5mm} 
	\infer{\Gamma \vdash \CL \Tree u~v~t: \Tree A \Rightarrow \Tree A}{\Gamma \vdash u : \Tree A}
	\vspace{5mm} 
	\infer{\Gamma \vdash \CR \Tree u~v~t: \Tree A \Rightarrow \Tree A}{\Gamma \vdash v : \Tree A}
	\vspace{5mm} 
	\infer{\Gamma \vdash \Node \Tree u~v~t: \Tree A \Rightarrow A}{\Gamma \vdash t : A}
	\vspace{5mm} 
	\infer{\Gamma \vdash \Node \Leaf t: \Leaf A \Rightarrow A}{\Gamma \vdash t : A}
	\vspace{5mm} 
	\infer{\Gamma \vdash \Tree u~v~t: \Tree A \Rightarrow \Tree A \Rightarrow A \Rightarrow\Tree A }
				{ \Gamma \vdash u : \Tree A 
				& \Gamma \vdash v : \Tree A
				& \Gamma \vdash t : A
				}
